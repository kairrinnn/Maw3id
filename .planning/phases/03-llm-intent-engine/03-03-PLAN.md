    ```typescript
    async function processMessage(
      supabase: ReturnType<typeof createServiceClient>,
      tenantId: string,
      message: WebhookMessage,
      phoneNumberId: string
    ) {
      const wamid = message.id
      const clientWaId = message.from

      // WA-02: Deduplicate via wamid PRIMARY KEY (unchanged from Phase 2)
      const { count } = await supabase
        .from('processed_messages')
        .upsert(
          { wamid, tenant_id: tenantId },
          { onConflict: 'wamid', ignoreDuplicates: true, count: 'exact' }
        )

      if (count === 0) {
        console.warn('[webhook] Duplicate wamid skipped:', wamid)
        return
      }

      // Upsert conversation (unchanged from Phase 2)
      await supabase
        .from('conversations')
        .upsert(
          {
            tenant_id: tenantId,
            wa_id: clientWaId,
            last_customer_message_at: new Date().toISOString(),
          },
          { onConflict: 'tenant_id,wa_id', ignoreDuplicates: false }
        )

      // Non-text messages: polite text-only reply (unchanged from Phase 2)
      if (message.type !== 'text') {
        await sendTextMessage(
          clientWaId,
          'Je ne peux traiter que les messages texte pour le moment.',
          phoneNumberId
        )
        return
      }

      // --- NEW: Full conversation flow (Phase 3) ---

      const userText = message.text.body

      // 1. Load tenant's active services
      const { data: services } = await supabase
        .from('services')
        .select('id, tenant_id, name, duration_minutes, price_mad, active')
        .eq('tenant_id', tenantId)
        .eq('active', true)

      const activeServices: Service[] = services ?? []

      // 2. Load conversation state from DB
      const state = await loadConversationState(supabase, tenantId, clientWaId)

      // 3. Extract intent via LLM (cheap -> escalation -> fallback)
      const intent = await extractIntent(userText, state, activeServices)

      // 4. Run FSM transition (pure function)
      const { nextState, replyText } = processIntent(
        state,
        intent,
        activeServices,
        userText  // confirmText for confirming step
      )

      // 5. Save updated state to DB
      await saveConversationState(supabase, tenantId, clientWaId, nextState)

      // 6. Send reply to client
      await sendTextMessage(clientWaId, replyText, phoneNumberId)
    }
    ```
