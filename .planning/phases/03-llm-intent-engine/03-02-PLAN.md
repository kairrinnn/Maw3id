export const FSM_STEPS = [
  'greeting', 'awaiting_service', 'awaiting_datetime',
  'confirming', 'confirmed', 'cancelled',
] as const

export interface IntentResult {
  service_name: string | null
  date_raw: string | null
  time_raw: string | null
  confidence: 'high' | 'low'
  intent: 'book' | 'cancel' | 'query' | 'greeting' | 'unknown'
}

export interface ConversationState {
  step: typeof FSM_STEPS[number]
  service_id?: string | null
  service_name?: string | null
  date?: string | null       // YYYY-MM-DD
  time?: string | null       // HH:MM
  client_name?: string | null
  status: 'idle' | 'in_progress' | 'done' | 'failed'
}

export interface Service {
  id: string; tenant_id: string; name: string;
  duration_minutes: number; price_mad: number | null; active: boolean
}

export const INITIAL_STATE: ConversationState = { step: 'greeting', status: 'idle' }

    **Create `src/lib/fsm/machine.ts`.**

    Export `processIntent(state: ConversationState, intent: IntentResult, services: Service[], confirmText?: string): { nextState: ConversationState; replyText: string }`:

    This is a PURE FUNCTION — no async, no DB calls, no LLM calls. It takes the current state and extracted intent, returns the next state and a French reply string.

    The `confirmText` parameter is the raw user message text, used only in the `confirming` step to detect affirmative/negative responses (e.g. "oui", "non", "wakha", "la"). For all other steps, it can be undefined.

    **Transition table implementation:**

    ```
    Step: greeting
      intent=book OR intent with service_name -> awaiting_service (or awaiting_datetime if service already matched)
      intent=greeting -> greeting, reply: "Bonjour! Comment puis-je vous aider? Vous pouvez reserver un rendez-vous."
      intent=cancel -> greeting, reply: "Vous n'avez pas de reservation en cours."
      intent=unknown -> greeting, reply: "Je n'ai pas compris. Vous pouvez me dire quel service vous souhaitez reserver."
      intent=query -> greeting, reply: "Pour consulter vos rendez-vous, tapez 'mes rdv'." (stub for Phase 4)

    Step: awaiting_service
      service_name non-null -> match against services list:
        - Exact case-insensitive match first
        - Then substring match (service.name.toLowerCase().includes(service_name.toLowerCase()))
        - If exactly 1 match: set service_id + service_name, advance to awaiting_datetime
        - If 0 matches: stay at awaiting_service, reply: "Je ne trouve pas ce service. Voici nos services: {list}"
        - If >1 match: stay at awaiting_service, reply: "Plusieurs services correspondent. Lequel souhaitez-vous? {matched names}"
      service_name null -> stay, reply: "Quel service souhaitez-vous? {service list}"

    Step: awaiting_datetime
      date_raw + time_raw both present -> resolve via resolveDate/resolveTime:
        - Both resolve: advance to confirming, set date+time, reply: "Vous souhaitez {service_name} le {date} a {time}. C'est bien ca? (oui/non)"
        - Either fails to resolve: stay, reply asking for the failed part
      date_raw only -> resolve, set date, reply: "A quelle heure souhaitez-vous votre rendez-vous?"
      time_raw only -> resolve, set time, reply: "Quel jour souhaitez-vous votre rendez-vous?"
      both null -> stay, reply: "Quand souhaitez-vous votre rendez-vous? (ex: demain a 15h)"

    Step: confirming
      Use confirmText (raw message) to detect:
        - Affirmative: /^(oui|yes|ok|wakha|waxxa|d'accord|daccord|ouais|yep|yah)$/i -> confirmed, status=done
          Reply: "Parfait! Votre rendez-vous est confirme: {service} le {date} a {time}."
        - Negative: /^(non|no|la|lala|annuler)$/i -> cancelled
          Reply: "D'accord, votre rendez-vous a ete annule."
        - Neither: stay at confirming, reply: "Veuillez repondre par oui ou non."

    Step: confirmed / cancelled
      Any intent -> reset to { step: "greeting", status: "idle" }, clear service/date/time
      Reply: "Bonjour! Comment puis-je vous aider?"
    ```

    **Global override:** If `intent.intent === 'cancel'` and step is NOT `confirmed` or `cancelled`, transition to `cancelled` regardless of current step. Reply: "D'accord, j'annule votre demande en cours."

    **Service matching helper** (not exported):
    ```typescript
    function matchService(serviceName: string, services: Service[]): Service | Service[] | null {
      // 1. Exact case-insensitive
      const exact = services.find(s => s.name.toLowerCase() === serviceName.toLowerCase())
      if (exact) return exact
      // 2. Substring
      const subs = services.filter(s => s.name.toLowerCase().includes(serviceName.toLowerCase())
        || serviceName.toLowerCase().includes(s.name.toLowerCase()))
      if (subs.length === 1) return subs[0]
      if (subs.length > 1) return subs
      return null
    }
    ```
