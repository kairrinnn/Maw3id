'use server'
import { z } from 'zod'
import { revalidatePath } from 'next/cache'
import { createClient } from '@/lib/supabase/server'
import { getOnboardingReadiness } from '@/lib/onboarding/readiness'

type ActionResult = { error: string | Record<string, string[]> } | { success: true }

async function getTenantId() {
  const supabase = await createClient()
  const { data: claimsData } = await supabase.auth.getClaims()
  const claims = claimsData?.claims as Record<string, unknown> | null
  const tenantId = (claims?.tenant_id ?? null) as string | null
  return { supabase, tenantId }
}

// ─── Schema definitions ────────────────────────────────────────────────────

const SalonProfileSchema = z.object({
  name: z.string().trim().min(1, 'Le nom est requis').max(100),
  description: z.string().trim().max(500).optional(),
})

const PhoneNumberSchema = z.object({
  phone_number_id: z.string().trim().min(5).max(100),
  waba_id: z.string().trim().min(5).max(100),
  display_phone: z.string().trim().min(8).max(20),
  connection_type: z.enum(['dedicated', 'coexistence']),
})

// ─── saveSalonProfile ──────────────────────────────────────────────────────

export async function saveSalonProfile(_prev: unknown, formData: FormData): Promise<ActionResult> {
  const { supabase, tenantId } = await getTenantId()
  if (!tenantId) return { error: 'Non autorisé' }

  const parsed = SalonProfileSchema.safeParse(Object.fromEntries(formData))
  if (!parsed.success) return { error: parsed.error.flatten().fieldErrors }

  const { error } = await supabase
    .from('tenants')
    .update({ name: parsed.data.name, description: parsed.data.description ?? null })
    .eq('id', tenantId)

  if (error) return { error: error.message }
  revalidatePath('/onboarding/1')
  return { success: true }
}

// ─── savePhoneNumber ───────────────────────────────────────────────────────

export async function savePhoneNumber(_prev: unknown, formData: FormData): Promise<ActionResult> {
  const { supabase, tenantId } = await getTenantId()
  if (!tenantId) return { error: 'Non autorisé' }

  const parsed = PhoneNumberSchema.safeParse(Object.fromEntries(formData))
  if (!parsed.success) return { error: parsed.error.flatten().fieldErrors }

  // Upsert — requires UNIQUE(tenant_id) from Wave 0 migration (42P10 if missing)
  const { error } = await supabase
    .from('phone_numbers')
    .upsert(
      {
        tenant_id: tenantId,
        phone_number_id: parsed.data.phone_number_id,
        waba_id: parsed.data.waba_id,
        display_phone: parsed.data.display_phone,
        status: 'pending',
      },
      { onConflict: 'tenant_id' }
    )

  if (error) return { error: error.message }
  revalidatePath('/onboarding/4')
  return { success: true }
}

// ─── activateBot ───────────────────────────────────────────────────────────

export async function activateBot(_prev: unknown, formData: FormData): Promise<ActionResult> {
  const { supabase, tenantId } = await getTenantId()
  if (!tenantId) return { error: 'Non autorisé' }

  const testMode = formData.get('test_mode') === 'on'

  // Guard 1: phone number row must exist
  const { data: phoneRow } = await supabase
    .from('phone_numbers')
    .select('id')
    .eq('tenant_id', tenantId)
    .maybeSingle()

  if (!phoneRow) return { error: "Connectez d'abord votre numéro WhatsApp" }

  // Guard 2: readiness gate (≥1 active service AND ≥1 open day)
  const readiness = await getOnboardingReadiness(tenantId)
  if (!readiness.ready) {
    const missing = !readiness.hasService ? 'aucun service' : 'aucun jour ouvert'
    return { error: `Configuration incomplète: ${missing}` }
  }

  const { error } = await supabase
    .from('bot_configs')
    .update({ active: true, test_mode: testMode })
    .eq('tenant_id', tenantId)

  if (error) return { error: error.message }
  revalidatePath('/dashboard')
  revalidatePath('/onboarding/4')
  return { success: true }
}

function makeFormData(fields: Record<string, string>): FormData {
  const fd = new FormData()
  for (const [k, v] of Object.entries(fields)) fd.append(k, v)
  return fd
}

function buildUpdateChain(err: null | { message: string } = null) {
  const chain: Record<string, unknown> = {}
  chain.update = vi.fn(() => chain)
  chain.eq = vi.fn(() => Promise.resolve({ error: err }))
  return chain
}

function buildUpsertChain(err: null | { message: string } = null) {
  const chain: Record<string, unknown> = {}
  chain.upsert = vi.fn(() => Promise.resolve({ error: err }))
  return chain
}

function buildCountChain(count: number | null) {
  const chain: Record<string, unknown> = {}
  let eqCount = 0
  chain.select = vi.fn(() => chain)
  chain.eq = vi.fn(() => {
    eqCount++
    if (eqCount >= 2) { eqCount = 0; return Promise.resolve({ count }) }
    return chain
  })
  return chain
}

function buildMaybySingleChain(data: { id: string } | null) {
  const chain: Record<string, unknown> = {}
  chain.select = vi.fn(() => chain)
  chain.eq = vi.fn(() => chain)
  chain.maybeSingle = vi.fn(() => Promise.resolve({ data }))
  return chain
}
